# With the Flow — Web App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a mobile-first static web app for navigating the 30-day meal plan, with bottom tab navigation, expandable recipe cards, checkable grocery lists, and full session restore.

**Architecture:** Static SPA — vanilla HTML/CSS/JS, no framework, no build step. Data extracted from existing HTML into JSON. Hosted on GitHub Pages.

**Tech Stack:** HTML5, CSS3, vanilla JavaScript, Node.js (for data extraction script only), cheerio (HTML parsing)

**Spec:** `docs/superpowers/specs/2026-04-01-meal-plan-webapp-design.md`

```

├── app.js                       # View switching, navigation, expand/collapse, session restore

docs/superpowers/
```

**Why `webapp/` subfolder:** Keeps web app files isolated from the existing PDF generation files. GitHub Pages can be pointed at `webapp/` folder. The extraction script lives in `scripts/` alongside the existing PDF generation scripts.

```javascript
// scripts/extract-data.js
// Parses meal-plan-v2.html and grocery-lists-v2.html into webapp/data.json

const fs = require('fs');
const path = require('path');
const cheerio = require('cheerio');

const ROOT = path.join(__dirname, '..');
const mealHtml = fs.readFileSync(path.join(ROOT, 'meal-plan-v2.html'), 'utf-8');
const groceryHtml = fs.readFileSync(path.join(ROOT, 'grocery-lists-v2.html'), 'utf-8');

const $ = cheerio.load(mealHtml);
const $g = cheerio.load(groceryHtml);

// ── Extract days ──
const days = [];
const dayOverviews = $('.day-overview');

dayOverviews.each((i, el) => {
  const $day = $(el);
  const dayNum = parseInt($day.find('.day-number').text().replace('Day ', ''), 10);
  const totalsText = $day.find('.day-totals').text();
  const phaseText = $day.find('.day-phase').text().trim();
  const phase = phaseText.includes('Luteal') ? 'luteal' : 'standard';
  const adrenalNote = $day.find('.day-adrenal-note').text().trim();

  // Parse totals: "1495 cal · 146g protein · 36g fiber"
  const totalsMatch = totalsText.match(/(\d+)\s*cal\s*·\s*(\d+)g\s*protein\s*·\s*(\d+)g\s*fiber/);
  const totals = totalsMatch
    ? { calories: parseInt(totalsMatch[1]), protein: parseInt(totalsMatch[2]), fiber: parseInt(totalsMatch[3]) }
    : { calories: 0, protein: 0, fiber: 0 };

  days.push({ day: dayNum, phase, totals, meals: [], adrenalNote });
});

// ── Extract meal cards ──
$('.meal-card').each((i, el) => {
  const $card = $(el);
  const labelText = $card.find('.meal-label').first().text().trim();
  // "Day 1 · Meal 1" or "Day 1 · Snack"
  const labelMatch = labelText.match(/Day\s+(\d+)\s*·\s*(.+)/);
  if (!labelMatch) return;

  const dayNum = parseInt(labelMatch[1], 10);
  const mealLabel = labelMatch[2].trim();

  const name = $card.find('.meal-name').text().trim();

  // Parse macro pills
  const pills = [];
  $card.find('.macro-pill').each((j, pill) => pills.push($(pill).text().trim()));
  // "496 cal", "62g protein", "16g fiber"
  const calMatch = pills[0]?.match(/(\d+)\s*cal/);
  const proMatch = pills[1]?.match(/(\d+)g\s*protein/);
  const fibMatch = pills[2]?.match(/(\d+)g\s*fiber/);
  const macros = {
    calories: calMatch ? parseInt(calMatch[1]) : 0,
    protein: proMatch ? parseInt(proMatch[1]) : 0,
    fiber: fibMatch ? parseInt(fibMatch[1]) : 0,
  };

  // Parse ingredients
  const ingredients = [];
  $card.find('.ingredient').each((j, ing) => {
    const $ing = $(ing);
    const amount = $ing.find('.ingredient-amount').text().trim();
    const ingMacros = $ing.find('.ingredient-macros').text().trim();
    // Get ingredient name: text before the " — " separator
    const nameSpan = $ing.find('> span').first();
    let ingName = nameSpan.text().trim();
    // Remove the amount portion (everything after " — ")
    const dashIdx = ingName.indexOf('—');
    if (dashIdx > -1) ingName = ingName.substring(0, dashIdx).trim();

    ingredients.push({ name: ingName, amount, macros: ingMacros });
  });

  // Parse instructions
  const instructions = $card.find('.instructions-text').text().trim();

  const meal = { label: mealLabel, name, macros, ingredients, instructions };

  // Find the matching day and add this meal
  const dayObj = days.find(d => d.day === dayNum);
  if (dayObj) dayObj.meals.push(meal);
});

// ── Extract grocery lists ──
const groceryLists = [];
$g('.grocery-page').each((i, el) => {
  const $page = $g(el);
  const weekTitle = $page.find('.grocery-week h2').text().trim();
  const weekMatch = weekTitle.match(/Week\s+(\d+)/);
  const weekNum = weekMatch ? parseInt(weekMatch[1]) : i + 1;
  const subtitle = $page.find('.grocery-week-subtitle').text().trim();

  const categories = [];
  $page.find('.grocery-section').each((j, sec) => {
    const $sec = $g(sec);
    const catName = $sec.find('.grocery-category').text().trim();
    const items = [];
    $sec.find('.grocery-item').each((k, item) => {
      items.push({ text: $g(item).text().trim() });
    });
    categories.push({ name: catName, items });
  });

  groceryLists.push({ week: weekNum, title: weekTitle, subtitle, categories });
});

// ── Extract info pages ──
const info = {};

// How to Use
const $howToUse = $('#how-to-use');
info.howToUse = {
  title: $howToUse.find('h2').text().trim(),
  intro: $howToUse.find('.intro-page-header > p').text().trim(),
  sections: [],
};
$howToUse.find('h3').each((i, el) => {
  const $h3 = $(el);
  const sectionTitle = $h3.text().trim();
  // Collect text until next h3 or phase-block
  const nextP = $h3.next('p');
  info.howToUse.sections.push({ title: sectionTitle, text: nextP.text().trim() });
});
// Phase blocks
info.howToUse.phases = [];
$howToUse.find('.phase-block').each((i, el) => {
  const $pb = $(el);
  info.howToUse.phases.push({
    label: $pb.find('.phase-label').text().trim(),
    title: $pb.find('.phase-title').text().trim(),
    description: $pb.find('p').text().trim(),
    mealsTag: $pb.find('.phase-meals-tag').text().trim(),
  });
});
// Callout
const $callout = $howToUse.find('.callout-box');
if ($callout.length) {
  info.howToUse.callout = {
    label: $callout.find('.callout-label').text().trim(),
    text: $callout.find('p').text().trim(),
  };
}

// Understanding Calories
const $calories = $('#understanding-calories');
info.understandingCalories = {
  title: $calories.find('h2').text().trim(),
  intro: $calories.find('.intro-page-header > p').text().trim(),
  rows: [],
};
$calories.find('.info-row').each((i, el) => {
  const $row = $(el);
  info.understandingCalories.rows.push({
    title: $row.find('h4').text().trim(),
    text: $row.find('p').text().trim(),
  });
});

// Why High Protein + High Fiber
const $whyPF = $('#why-protein-fiber');
info.whyProteinFiber = {
  title: $whyPF.find('h2').text().trim(),
  intro: $whyPF.find('.intro-page-header > p').text().trim(),
  proteinSection: $whyPF.find('.nutrient-block').first().find('p').text().trim(),
  fiberSection: $whyPF.find('.nutrient-block').last().find('p').text().trim(),
};

// Adrenal Cocktail Recipe
const $ac = $('#adrenal-cocktail');
info.adrenalCocktail = {
  title: 'Adrenal Cocktail',
  ingredients: [],
  macrosNote: '',
};
$ac.find('.recipe-ingredients li').each((i, el) => {
  info.adrenalCocktail.ingredients.push($(el).text().trim());
});
info.adrenalCocktail.macrosNote = $ac.find('.recipe-macros-note').text().trim();

// ── Validation ──
const errors = [];

// Check day count
if (days.length !== 30) errors.push(`Expected 30 days, found ${days.length}`);

// Check meals per day
days.forEach(d => {
  const expectedMeals = d.phase === 'standard' ? 3 : 4; // meals + snack
  if (d.meals.length !== expectedMeals) {
    errors.push(`Day ${d.day}: expected ${expectedMeals} meals, found ${d.meals.length}`);
  }

  // Check macro sums
  const sumCal = d.meals.reduce((s, m) => s + m.macros.calories, 0);
  const sumPro = d.meals.reduce((s, m) => s + m.macros.protein, 0);
  const sumFib = d.meals.reduce((s, m) => s + m.macros.fiber, 0);

  // Note: day totals include adrenal cocktails (320 cal, 8g protein)
  // So meal sum + 320 should equal day total calories
  // And meal sum + 8 should equal day total protein
  // Fiber from cocktails is 0

  // Allow some tolerance for rounding
  if (Math.abs((sumCal + 320) - d.totals.calories) > 5) {
    errors.push(`Day ${d.day}: calorie mismatch. Meals sum ${sumCal} + 320 cocktail = ${sumCal + 320}, expected ${d.totals.calories}`);
  }
  if (Math.abs((sumPro + 8) - d.totals.protein) > 5) {
    errors.push(`Day ${d.day}: protein mismatch. Meals sum ${sumPro} + 8 cocktail = ${sumPro + 8}, expected ${d.totals.protein}`);
  }

  // Check each meal has ingredients and instructions
  d.meals.forEach(m => {
    if (m.ingredients.length === 0) errors.push(`Day ${d.day} ${m.label}: no ingredients found`);
    if (!m.instructions) errors.push(`Day ${d.day} ${m.label}: no instructions found`);
  });
});

// Check grocery lists
if (groceryLists.length !== 4) errors.push(`Expected 4 grocery weeks, found ${groceryLists.length}`);
groceryLists.forEach(gl => {
  if (gl.categories.length === 0) errors.push(`Week ${gl.week}: no categories found`);
  gl.categories.forEach(cat => {
    if (cat.items.length === 0) errors.push(`Week ${gl.week} ${cat.name}: no items found`);
  });
});

// ── Output ──
if (errors.length > 0) {
  console.error('\n❌ VALIDATION ERRORS:');
  errors.forEach(e => console.error(`  ✗ ${e}`));
  console.error(`\n${errors.length} error(s) found. Fix before proceeding.\n`);
  process.exit(1);
}

const data = { days, groceryLists, info };
const outPath = path.join(ROOT, 'webapp', 'data.json');
fs.mkdirSync(path.join(ROOT, 'webapp'), { recursive: true });
fs.writeFileSync(outPath, JSON.stringify(data, null, 2));

// Print validation report
console.log('\n✅ EXTRACTION COMPLETE');
console.log(`   ${days.length} days extracted`);
days.forEach(d => {
  console.log(`   ✓ Day ${d.day} (${d.phase}): ${d.meals.length} meals, ${d.totals.calories} cal / ${d.totals.protein}g P / ${d.totals.fiber}g Fi`);
});
console.log(`   ${groceryLists.length} grocery weeks extracted`);
groceryLists.forEach(gl => {
  const totalItems = gl.categories.reduce((s, c) => s + c.items.length, 0);
  console.log(`   ✓ Week ${gl.week}: ${gl.categories.length} categories, ${totalItems} items`);
});
console.log(`\n   Output: ${outPath}\n`);
```

```javascript
// ============================================================
// WITH THE FLOW — Web App
// View switching, day navigation, recipe expand, grocery checks,
// session restore
// ============================================================

(function () {
  'use strict';

  let DATA = null;
  let state = {
    activeView: 'today',
    currentDay: 1,
    activeWeek: 1,
    expandedMeals: [],   // ["1-0", "1-2"] = day 1 meal index 0 and 2
    groceryChecked: {},   // { "1-Protein-0": true }
    scrollPositions: {},  // { "today": 120, "grocery": 0 }
  };

  const STORAGE_KEY = 'wtf-app-state';

  // ── Init ──
  async function init() {
    const resp = await fetch('data.json');
    DATA = await resp.json();
    loadState();
    renderDayPicker();
    renderDay(state.currentDay);
    renderAllDays();
    renderGrocery(state.activeWeek);
    renderInfo();
    bindTabs();
    switchView(state.activeView);
    restoreScrollPositions();
  }

  // ── State persistence ──
  function saveState() {
    // Capture current scroll position
    const activeViewEl = document.querySelector('.view.active .view-content');
    if (activeViewEl) {
      state.scrollPositions[state.activeView] = activeViewEl.scrollTop;
    }
    localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
  }

  function loadState() {
    try {
      const saved = localStorage.getItem(STORAGE_KEY);
      if (saved) {
        const parsed = JSON.parse(saved);
        state = { ...state, ...parsed };
      }
    } catch (e) {
      // Ignore corrupt state
    }
  }

  function restoreScrollPositions() {
    Object.entries(state.scrollPositions).forEach(([view, pos]) => {
      const el = document.querySelector(`#view-${view} .view-content`);
      if (el) el.scrollTop = pos;
    });
  }

  // ── Tab switching ──
  function bindTabs() {
    document.querySelectorAll('.tab').forEach(tab => {
      tab.addEventListener('click', () => {
        const view = tab.dataset.view;
        switchView(view);
      });
    });
  }

  function switchView(viewName) {
    // Save scroll position of current view before switching
    const currentContent = document.querySelector('.view.active .view-content');
    if (currentContent) {
      state.scrollPositions[state.activeView] = currentContent.scrollTop;
    }

    state.activeView = viewName;

    // Update views
    document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
    document.getElementById(`view-${viewName}`).classList.add('active');

    // Update tabs
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    document.querySelector(`.tab[data-view="${viewName}"]`).classList.add('active');

    // Update header
    updateHeader(viewName);

    // Restore scroll position for new view
    const newContent = document.querySelector(`#view-${viewName} .view-content`);
    if (newContent && state.scrollPositions[viewName]) {
      newContent.scrollTop = state.scrollPositions[viewName];
    }

    saveState();
  }

  function updateHeader(viewName) {
    const title = document.getElementById('header-title');
    const subtitle = document.getElementById('header-subtitle');

    if (viewName === 'today') {
      const day = DATA.days.find(d => d.day === state.currentDay);
      title.textContent = `Day ${state.currentDay}`;
      const phaseLabel = day.phase === 'luteal' ? 'Luteal Phase · Days 21–30' : 'Standard Phase · Days 1–20';
      subtitle.textContent = phaseLabel;
      subtitle.className = 'phase-badge' + (day.phase === 'luteal' ? ' luteal' : '');
    } else if (viewName === 'all-days') {
      title.textContent = 'All Days';
      subtitle.textContent = '30-day overview';
      subtitle.className = 'phase-badge';
    } else if (viewName === 'grocery') {
      title.textContent = 'Grocery Lists';
      subtitle.textContent = 'Shop by week';
      subtitle.className = 'phase-badge';
    } else if (viewName === 'info') {
      title.textContent = 'Guide';
      subtitle.textContent = 'How it works';
      subtitle.className = 'phase-badge';
    }
  }

  // ── Day Picker ──
  function renderDayPicker() {
    const picker = document.getElementById('day-picker');
    picker.innerHTML = DATA.days.map(d => {
      const isActive = d.day === state.currentDay ? ' active' : '';
      const isLuteal = d.phase === 'luteal' ? ' luteal' : '';
      return `<div class="day-chip${isLuteal}${isActive}" data-day="${d.day}">
        <span style="font-size:9px">D</span>
        <span class="day-chip-num">${d.day}</span>
      </div>`;
    }).join('');

    picker.addEventListener('click', e => {
      const chip = e.target.closest('.day-chip');
      if (!chip) return;
      const dayNum = parseInt(chip.dataset.day, 10);
      selectDay(dayNum);
    });
  }

  function selectDay(dayNum) {
    state.currentDay = dayNum;
    state.expandedMeals = [];

    // Update chips
    document.querySelectorAll('.day-chip').forEach(c => c.classList.remove('active'));
    document.querySelector(`.day-chip[data-day="${dayNum}"]`).classList.add('active');

    // Scroll chip into view
    const activeChip = document.querySelector(`.day-chip[data-day="${dayNum}"]`);
    activeChip.scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'center' });

    renderDay(dayNum);
    updateHeader('today');
    saveState();
  }

  // ── Day View ──
  function renderDay(dayNum) {
    const day = DATA.days.find(d => d.day === dayNum);
    const content = document.getElementById('day-content');

    const macrosHtml = `
      <div class="day-macros">
        <div class="macro-item"><div class="macro-val">${day.totals.calories}</div><div class="macro-label">Calories</div></div>
        <div class="macro-item"><div class="macro-val">${day.totals.protein}g</div><div class="macro-label">Protein</div></div>
        <div class="macro-item"><div class="macro-val">${day.totals.fiber}g</div><div class="macro-label">Fiber</div></div>
      </div>`;

    const mealsHtml = day.meals.map((meal, idx) => {
      const isSnack = meal.label.toLowerCase().includes('snack');
      const snackClass = isSnack ? ' snack' : '';
      const mealKey = `${dayNum}-${idx}`;
      const isExpanded = state.expandedMeals.includes(mealKey);

      const ingredientsHtml = meal.ingredients.map(ing =>
        `<div class="ingredient-row">
          <span>${escapeHtml(ing.name)}</span>
          <span class="ingredient-amount">${escapeHtml(ing.amount)}</span>
        </div>
        <div class="ingredient-macros-text">${escapeHtml(ing.macros)}</div>`
      ).join('');

      return `
        <div class="meal-card${snackClass}">
          <div class="meal-tag">${escapeHtml(meal.label)}</div>
          <div class="meal-name">${escapeHtml(meal.name)}</div>
          <div class="meal-macros">
            <span class="meal-macro-pill">${meal.macros.calories} cal</span>
            <span class="meal-macro-pill">${meal.macros.protein}g P</span>
            <span class="meal-macro-pill">${meal.macros.fiber}g Fi</span>
          </div>
          <button class="expand-btn${isExpanded ? ' expanded' : ''}" data-meal="${mealKey}">
            <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="6 9 12 15 18 9"></polyline></svg>
            ${isExpanded ? 'Hide Recipe' : 'View Recipe'}
          </button>
          <div class="recipe-expanded${isExpanded ? ' open' : ''}">
            <div class="recipe-section-label">Ingredients</div>
            ${ingredientsHtml}
            <div class="recipe-section-label" style="margin-top:16px">How to Make It</div>
            <div class="instructions-text">${escapeHtml(meal.instructions)}</div>
          </div>
        </div>`;
    }).join('');

    content.innerHTML = macrosHtml + mealsHtml +
      `<div class="adrenal-note">${escapeHtml(day.adrenalNote)}</div>`;

    // Bind expand buttons
    content.querySelectorAll('.expand-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const key = btn.dataset.meal;
        const recipe = btn.closest('.meal-card').querySelector('.recipe-expanded');
        const isOpen = recipe.classList.contains('open');

        if (isOpen) {
          recipe.classList.remove('open');
          btn.classList.remove('expanded');
          btn.innerHTML = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="6 9 12 15 18 9"></polyline></svg> View Recipe';
          state.expandedMeals = state.expandedMeals.filter(k => k !== key);
        } else {
          recipe.classList.add('open');
          btn.classList.add('expanded');
          btn.innerHTML = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="6 9 12 15 18 9"></polyline></svg> Hide Recipe';
          state.expandedMeals.push(key);
        }
        saveState();
      });
    });
  }

  // ── All Days Grid ──
  function renderAllDays() {
    const grid = document.getElementById('all-days-grid');
    let html = '<div class="all-days-grid">';
    let lastPhase = '';

    DATA.days.forEach(d => {
      if (d.phase !== lastPhase) {
        if (d.phase === 'luteal') {
          html += '<div class="phase-divider">Luteal Phase · Days 21–30</div>';
        }
        lastPhase = d.phase;
      }

      const mealNames = d.meals.map(m => `<li>${escapeHtml(m.name)}</li>`).join('');
      html += `
        <div class="all-day-card ${d.phase}" data-day="${d.day}">
          <div class="all-day-num">Day ${d.day}</div>
          <div class="all-day-phase">${d.phase === 'luteal' ? 'Luteal' : 'Standard'}</div>
          <ul class="all-day-meals">${mealNames}</ul>
        </div>`;
    });

    html += '</div>';
    grid.innerHTML = html;

    // Bind clicks
    grid.querySelectorAll('.all-day-card').forEach(card => {
      card.addEventListener('click', () => {
        const dayNum = parseInt(card.dataset.day, 10);
        selectDay(dayNum);
        switchView('today');
      });
    });
  }

  // ── Grocery View ──
  function renderGrocery(weekNum) {
    const weekData = DATA.groceryLists.find(gl => gl.week === weekNum);
    if (!weekData) return;

    // Render week tabs
    const tabsEl = document.getElementById('week-tabs');
    tabsEl.innerHTML = DATA.groceryLists.map(gl =>
      `<button class="week-tab${gl.week === weekNum ? ' active' : ''}" data-week="${gl.week}">Week ${gl.week}</button>`
    ).join('');

    tabsEl.querySelectorAll('.week-tab').forEach(tab => {
      tab.addEventListener('click', () => {
        const w = parseInt(tab.dataset.week, 10);
        state.activeWeek = w;
        renderGrocery(w);
        saveState();
      });
    });

    // Render grocery items
    const content = document.getElementById('grocery-content');
    let html = '';

    weekData.categories.forEach(cat => {
      html += `<div class="grocery-category">${escapeHtml(cat.name)}</div>`;
      cat.items.forEach((item, idx) => {
        const checkKey = `${weekNum}-${cat.name}-${idx}`;
        const isChecked = state.groceryChecked[checkKey] || false;
        html += `
          <div class="grocery-item${isChecked ? ' checked-item' : ''}">
            <button class="grocery-check${isChecked ? ' checked' : ''}" data-key="${checkKey}">${isChecked ? '✓' : ''}</button>
            <span class="grocery-item-text">${escapeHtml(item.text)}</span>
          </div>`;
      });
    });

    content.innerHTML = html;

    // Bind checkboxes
    content.querySelectorAll('.grocery-check').forEach(btn => {
      btn.addEventListener('click', () => {
        const key = btn.dataset.key;
        const isChecked = !state.groceryChecked[key];
        state.groceryChecked[key] = isChecked;

        btn.classList.toggle('checked', isChecked);
        btn.textContent = isChecked ? '✓' : '';
        btn.closest('.grocery-item').classList.toggle('checked-item', isChecked);
        saveState();
      });
    });
  }

  // ── Info View ──
  function renderInfo() {
    const content = document.getElementById('info-content');
    const info = DATA.info;
    let html = '';

    // How to Use
    if (info.howToUse) {
      html += '<div class="info-section">';
      html += `<h2>${escapeHtml(info.howToUse.title)}</h2>`;
      if (info.howToUse.intro) html += `<p>${escapeHtml(info.howToUse.intro)}</p>`;

      if (info.howToUse.sections) {
        info.howToUse.sections.forEach(s => {
          html += `<h3>${escapeHtml(s.title)}</h3>`;
          if (s.text) html += `<p>${escapeHtml(s.text)}</p>`;
        });
      }

      if (info.howToUse.phases) {
        info.howToUse.phases.forEach(p => {
          const isLuteal = p.title.toLowerCase().includes('luteal');
          html += `<div class="info-phase-block${isLuteal ? ' luteal' : ''}">
            <div class="info-phase-label">${escapeHtml(p.label)}</div>
            <h3>${escapeHtml(p.title)}</h3>
            <p>${escapeHtml(p.description)}</p>
            <p style="font-size:11px;color:var(--text-muted);margin-top:6px">${escapeHtml(p.mealsTag)}</p>
          </div>`;
        });
      }

      if (info.howToUse.callout) {
        html += `<div class="info-callout">
          <strong>${escapeHtml(info.howToUse.callout.label)}</strong>
          <p>${escapeHtml(info.howToUse.callout.text)}</p>
        </div>`;
      }
      html += '</div>';
    }

    // Understanding Calories
    if (info.understandingCalories) {
      html += '<div class="info-section">';
      html += `<h2>${escapeHtml(info.understandingCalories.title)}</h2>`;
      if (info.understandingCalories.intro) html += `<p>${escapeHtml(info.understandingCalories.intro)}</p>`;
      if (info.understandingCalories.rows) {
        info.understandingCalories.rows.forEach(r => {
          html += `<h3>${escapeHtml(r.title)}</h3><p>${escapeHtml(r.text)}</p>`;
        });
      }
      html += '</div>';
    }

    // Why High Protein + Fiber
    if (info.whyProteinFiber) {
      html += '<div class="info-section">';
      html += `<h2>${escapeHtml(info.whyProteinFiber.title)}</h2>`;
      if (info.whyProteinFiber.intro) html += `<p>${escapeHtml(info.whyProteinFiber.intro)}</p>`;
      if (info.whyProteinFiber.proteinSection) html += `<h3>Protein</h3><p>${escapeHtml(info.whyProteinFiber.proteinSection)}</p>`;
      if (info.whyProteinFiber.fiberSection) html += `<h3>Fiber</h3><p>${escapeHtml(info.whyProteinFiber.fiberSection)}</p>`;
      html += '</div>';
    }

    // Adrenal Cocktail
    if (info.adrenalCocktail) {
      html += '<div class="info-recipe-card">';
      html += `<h2>${escapeHtml(info.adrenalCocktail.title)}</h2>`;
      html += '<ul>';
      info.adrenalCocktail.ingredients.forEach(ing => {
        html += `<li>${escapeHtml(ing)}</li>`;
      });
      html += '</ul>';
      if (info.adrenalCocktail.macrosNote) {
        html += `<p style="margin-top:12px;font-size:12px;color:var(--text-muted);text-align:center">${escapeHtml(info.adrenalCocktail.macrosNote)}</p>`;
      }
      html += '</div>';
    }

    content.innerHTML = html;
  }

  // ── Utilities ──
  function escapeHtml(str) {
    if (!str) return '';
    const div = document.createElement('div');
    div.textContent = str;
    return div.innerHTML;
  }

  // ── Save state on scroll (debounced) ──
  let scrollTimer;
  document.addEventListener('scroll', () => {
    clearTimeout(scrollTimer);
    scrollTimer = setTimeout(saveState, 300);
  }, true);

  // Save state before page unload
  window.addEventListener('beforeunload', saveState);

  // ── Start ──
  document.addEventListener('DOMContentLoaded', init);
})();
```
